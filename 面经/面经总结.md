1、大端小端的区别 应用 如何判断大端小端

- 如何判断

- ```cpp
  #include<iostream>
  
  using namespace std;
  
  
  int main(){
      union{
  		int n;
  		char ch;
  	} data;
  	data.n = 0x00000001; 
      //union下公用一段四字节地址空间，data.ch只占一个字节，小端情况下为01，大端情况为00
      cout << data.n << endl;
      cout << data.ch << endl;;
  	if(data.ch == 1){
  		cout << "little" << endl;
  	}
  	else{
  		cout << "big" << endl;
  	}
  	return 0;
  }
  ```
